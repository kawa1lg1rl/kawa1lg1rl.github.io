# 5. 함수

```golang
package main

import (
	"fmt"

)

// 자료형이 같을경우 생략가능
// int a,b 같은 느낌
func test(a, b int) int {
	return a + b 
}
func main() {
	fmt.Println( test(1, 2) )
}	

```

```golang
package main

import (
	"fmt"

)

// 여러개 return
func test(a int, b int) (int , int) { 
	return a , b 
}
func main() {
    a := 1
    b := 2
    fmt.Println(a, b)

    b, a = test(a, b)
    fmt.Println(a, b)
}
```


```golang
package main

import (
	"fmt"

)

// return값 이름 적어주기 
func test(a int, b int) (x, y int) { 
	x, y = b, a 
	return x , y 
}
func main() {
    a := 1
    b := 2
    fmt.Println(a, b)
    
    a, b = test(a, b)
    fmt.Println(a, b)
}	

```




# 6. for

```golang
package main

import (
	"fmt"
)

func main() {
    
    for i := 0; i < 10; i++ {
        fmt.Println(i)
    }
    
}

```
